def nextSong(url, num = 0):
    r = requests.get(url)
    soup = BeautifulSoup(r.text, "html.parser")
    val = soup.select("a")
    m = re.search('href="/watch\?v=(.{11})"', str(val[20 + num]))
    if(m == None):
        return nextSong(url, num+1)
    return 'https://www.youtube.com/watch?v=' + m.group(0)[15:-1]```
